310 static inline bool
311 compare_pickInterest(const pit::InRecord& a, const pit::InRecord& b, const Face* outFace)
312 {
313  bool isOutFaceA = a.getFace().get() == outFace;
314  bool isOutFaceB = b.getFace().get() == outFace;
315 
316  if (!isOutFaceA && isOutFaceB) {
317  return false;
318  }
319  if (isOutFaceA && !isOutFaceB) {
320  return true;
321  }
322 
323  return a.getLastRenewed() > b.getLastRenewed();

327 Forwarder::onOutgoingInterest(shared_ptr<pit::Entry> pitEntry, Face& outFace,
328  bool wantNewNonce)

330  if (outFace.getId() == INVALID_FACEID) {
331  NFD_LOG_WARN("onOutgoingInterest face=invalid interest=" << pitEntry->getName());
332  return;
333  }
334  NFD_LOG_DEBUG("onOutgoingInterest face=" << outFace.getId() <<
335  " interest=" << pitEntry->getName());
336 
337  // scope control

339  NFD_LOG_DEBUG("onOutgoingInterest face=" << outFace.getId() <<

344  // pick Interest
345  const pit::InRecordCollection& inRecords = pitEntry->getInRecords();
346  pit::InRecordCollection::const_iterator pickedInRecord = std::max_element(

348  BOOST_ASSERT(pickedInRecord != inRecords.end());
349  shared_ptr<Interest> interest = const_pointer_cast<Interest>(
350  pickedInRecord->getInterest().shared_from_this());
351 
352  if (wantNewNonce) {

355  interest->setNonce(dist(getGlobalRng()));
356  }
357 
358  // insert OutRecord
359  pitEntry->insertOrUpdateOutRecord(outFace.shared_from_this(), *interest);
360 
361  // send Interest

415 void
416 Forwarder::onIncomingData(Face& inFace, const Data& data)
417 {
418  // receive Data
419  NFD_LOG_DEBUG("onIncomingData face=" << inFace.getId() << " data=" << data.getName());
420  const_cast<Data&>(data).setIncomingFaceId(inFace.getId());
421  ++m_counters.getNInDatas();
422 
423  // /localhost scope control
424  bool isViolatingLocalhost = !inFace.isLocal() &&
425  LOCALHOST_NAME.isPrefixOf(data.getName());
426  if (isViolatingLocalhost) {
427  NFD_LOG_DEBUG("onIncomingData face=" << inFace.getId() <<
428  " data=" << data.getName() << " violates /localhost");
429  // (drop)
430  return;
431  }
432 
433  // PIT match
434  pit::DataMatchResult pitMatches = m_pit.findAllDataMatches(data);
435  pit::SitDataMatchResult sitMatches = m_sit.findAllDataMatches(data);
436  if (pitMatches.begin() == pitMatches.end() && sitMatches.begin() == sitMatches.end()) {
437  // goto Data unsolicited pipeline
438  this->onDataUnsolicited(inFace, data);
439  return;
440  }
441 
442  // Remove Ptr<Packet> from the Data before inserting into cache, serving two purposes
443  // - reduce amount of memory used by cached entries
444  // - remove all tags that (e.g., hop count tag) that could have been associated with Ptr<Packet>
445  //
446  // Copying of Data is relatively cheap operation, as it copies (mostly) a collection of Blocks
447  // pointing to the same underlying memory buffer.
448  shared_ptr<Data> dataCopyWithoutPacket = make_shared<Data>(data);
449  dataCopyWithoutPacket->removeTag<ns3::ndn::Ns3PacketTag>();
450 
451  // CS insert
452  if (m_csFromNdnSim == nullptr)
453  m_cs.insert(*dataCopyWithoutPacket);
454  else
455  m_csFromNdnSim->Add(dataCopyWithoutPacket);
456 
457  std::set<shared_ptr<Face> > pendingDownstreams;
458 
459  // foreach PitEntry
460  for (const shared_ptr<pit::Entry>& pitEntry : pitMatches) {
461  NFD_LOG_DEBUG("onIncomingData matching=" << pitEntry->getName());
462 
463  // cancel unsatisfy & straggler timer
464  this->cancelUnsatisfyAndStragglerTimer(pitEntry);
465 
466  // remember pending downstreams
467  const pit::InRecordCollection& inRecords = pitEntry->getInRecords();
468  for (pit::InRecordCollection::const_iterator it = inRecords.begin();
469  it != inRecords.end(); ++it) {
470  if (it->getExpiry() > time::steady_clock::now()) {
471  pendingDownstreams.insert(it->getFace());
472  }
473  }
474 
475  // invoke PIT satisfy callback
476  beforeSatisfyInterest(*pitEntry, inFace, data);
477  this->dispatchToStrategy(pitEntry, bind(&Strategy::beforeSatisfyInterest, _1,
478  pitEntry, cref(inFace), cref(data)));
479 
480  // Dead Nonce List insert if necessary (for OutRecord of inFace)
481  this->insertDeadNonceList(*pitEntry, true, data.getFreshnessPeriod(), &inFace);
482 
483  // mark PIT satisfied
484  pitEntry->deleteInRecords();
485  pitEntry->deleteOutRecord(inFace);
486 
487  // set PIT straggler timer
488  this->setStragglerTimer(pitEntry, true, data.getFreshnessPeriod());
489  }
490 
491  // foreach SitEntry
492  for (const shared_ptr<pit::Entry>& pitEntry : sitMatches) {
493  NFD_LOG_DEBUG("onIncomingData matching=" << pitEntry->getName());
494 
495  // cancel unsatisfy & straggler timer
496  this->cancelUnsatisfyAndStragglerTimer(pitEntry);
497 
498  // remember pending downstreams
499  const pit::InRecordCollection& inRecords = pitEntry->getInRecords();
500  for (pit::InRecordCollection::const_iterator it = inRecords.begin();
501  it != inRecords.end(); ++it) {
502  if (it->getExpiry() > time::steady_clock::now()) {
503  pendingDownstreams.insert(it->getFace());
504  }
505  }
506 
507  // invoke SIT satisfy callback
508  beforeSatisfyInterest(*pitEntry, inFace, data);
509  this->dispatchToStrategy(pitEntry, bind(&Strategy::beforeSatisfyInterest, _1,
510  pitEntry, cref(inFace), cref(data)));
511 
512  // Dead Nonce List insert if necessary (for OutRecord of inFace)
513  this->insertDeadNonceList(*pitEntry, true, data.getFreshnessPeriod(), &inFace);
514 
515  // mark PIT satisfied
516  //pitEntry->deleteInRecords();
517  //pitEntry->deleteOutRecord(inFace);
518 
519  // set PIT straggler timer
520  this->setStragglerTimer(pitEntry, true, data.getFreshnessPeriod());
521  }
522 
523  // foreach pending downstream
524  for (std::set<shared_ptr<Face> >::iterator it = pendingDownstreams.begin();
525  it != pendingDownstreams.end(); ++it) {
526  shared_ptr<Face> pendingDownstream = *it;
527  if (pendingDownstream.get() == &inFace) {
528  continue;
529  }
530  // goto outgoing Data pipeline
531  this->onOutgoingData(data, *pendingDownstream);
532  }
533 }

629 void
630 Forwarder::insertDeadNonceList(pit::Entry& pitEntry, bool isSatisfied,
631  const time::milliseconds& dataFreshnessPeriod,
632  Face* upstream)
633 {
634  // need Dead Nonce List insert?
635  bool needDnl = false;
636  if (isSatisfied) {
637  bool hasFreshnessPeriod = dataFreshnessPeriod >= time::milliseconds::zero();
638  // Data never becomes stale if it doesn't have FreshnessPeriod field
639  needDnl = static_cast<bool>(pitEntry.getInterest().getMustBeFresh()) &&
640  (hasFreshnessPeriod && dataFreshnessPeriod < m_deadNonceList.getLifetime());
641  }
642  else {
643  needDnl = true;
644  }
645 
646  if (!needDnl) {
647  return;
648  }
649 
650  // Dead Nonce List insert
651  if (upstream == 0) {
652  // insert all outgoing Nonces
653  const pit::OutRecordCollection& outRecords = pitEntry.getOutRecords();
654  std::for_each(outRecords.begin(), outRecords.end(),
655  bind(&insertNonceToDnl, ref(m_deadNonceList), cref(pitEntry), _1));
656  }
657  else {
658  // insert outgoing Nonce of a specific face
659  pit::OutRecordCollection::const_iterator outRecord = pitEntry.getOutRecord(*upstream);
660  if (outRecord != pitEntry.getOutRecords().end()) {
661  m_deadNonceList.add(pitEntry.getName(), outRecord->getLastNonce());
662  }
663  }
664 }
