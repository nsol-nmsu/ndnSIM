140 Ptr<FaceContainer>
141 StackHelper::Install(Ptr<Node> node) const
142 {
143  Ptr<FaceContainer> faces = Create<FaceContainer>();
144 
145  if (node->GetObject<L3Protocol>() != 0) {
146  NS_FATAL_ERROR("Cannot re-install NDN stack on node "
147  << node->GetId());
148  return 0;
149  }
150 
151  Ptr<L3Protocol> ndn = m_ndnFactory.Create<L3Protocol>();
152  ndn->getConfig().put("tables.cs_max_packets", (m_maxCsSize == 0) ? 1 : m_maxCsSize);
153 
154  // Create and aggregate content store if NFD's contest store has been disabled
155  if (m_maxCsSize == 0) {
156  ndn->AggregateObject(m_contentStoreFactory.Create<ContentStore>());
157  }
158 
159  // Aggregate L3Protocol on node (must be after setting ndnSIM CS)
160  node->AggregateObject(ndn);
161 
162  for (uint32_t index = 0; index < node->GetNDevices(); index++) {
163  Ptr<NetDevice> device = node->GetDevice(index);
164  // This check does not make sense: LoopbackNetDevice is installed only if IP stack is installed,
165  // Normally, ndnSIM works without IP stack, so no reason to check
166  // if (DynamicCast<LoopbackNetDevice> (device) != 0)
167  // continue; // don't create face for a LoopbackNetDevice
168 
169  faces->Add(this->createAndRegisterFace(node, ndn, device));
170  }
171 
172  return faces;
173 }

240 void
241 StackHelper::Update(Ptr<Node> node)
242 {
243  if (node->GetObject<L3Protocol>() == 0) {
244  Install(node);
245  return;
246  }
247 
248  Ptr<L3Protocol> ndn = node->GetObject<L3Protocol>();
249 
250  for (uint32_t index = 0; index < node->GetNDevices(); index++) {
251

281 shared_ptr<NetDeviceFace>
282 StackHelper::createAndRegisterFace(Ptr<Node> node, Ptr<L3Protocol> ndn, Ptr<NetDevice> device) const
283 {
284  shared_ptr<NetDeviceFace> face;
285 
286  for (const auto& item : m_netDeviceCallbacks) {
287  if (device->GetInstanceTypeId() == item.first ||
288  device->GetInstanceTypeId().IsChildOf(item.first)) {
289  face = item.second(node, ndn, device);
290  if (face != 0)
291  break;
292  }
293  }
294 
295  if (face == 0) {
296  face = DefaultNetDeviceCallback(node, ndn, device);
297  }
298 
299  if (m_needSetDefaultRoutes) {
300  // default route with lowest priority possible
301  FibHelper::AddRoute(node, "/", face, std::numeric_limits<int32_t>::max());
302  }
303  return face;
304 }
