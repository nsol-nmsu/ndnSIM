166 void
167 Subscriber::CheckRetxTimeout()
168 {
169  Time now = Simulator::Now();
170 
171  Time rto = m_rtt->RetransmitTimeout();
172  // NS_LOG_DEBUG ("Current RTO: " << rto.ToDouble (Time::S) << "s");
173 
174  while (!m_seqTimeouts.empty()) {
175  SeqTimeoutsContainer::index<i_timestamp>::type::iterator entry =
176  m_seqTimeouts.get<i_timestamp>().begin();
177  if (entry->time + rto <= now) // timeout expired?
178  {
179  uint32_t seqNo = entry->seq;
180  m_seqTimeouts.get<i_timestamp>().erase(entry);
181  OnTimeout(seqNo);
182  }
183  else
184  break; // nothing else to do. All later packets need not be retransmitted
185  }
186 
187  m_retxEvent = Simulator::Schedule(m_retxTimer, &Subscriber::CheckRetxTimeout, this);
188 }

208 void
209 Subscriber::SendPacket()
210 {
211 
212  //Set default size for payload interets
213  if (m_subscription == 0 && m_virtualPayloadSize == 0)
214  m_virtualPayloadSize = 4;
215 
216  if (!m_active)
217  return;
218 
219  NS_LOG_FUNCTION_NOARGS();
220 
221  uint32_t seq = std::numeric_limits<uint32_t>::max();
222 
223  while (m_retxSeqs.size()) {
224  seq = *m_retxSeqs.begin();
225  m_retxSeqs.erase(m_retxSeqs.begin());
226  break;
227  }
228 
229  if (m_seqMax != std::numeric_limits<uint32_t>::max()) {
230  if (m_seq >= m_seqMax) {
231  return;
232  }
233  }
234 
235  seq = m_seq++;
236 
237  uint8_t payload[1] = {1};
238 
239  //
240  shared_ptr<Name> nameWithSequence = make_shared<Name>(m_interestName);
241  //
242 
243  shared_ptr<Interest> interest = make_shared<Interest>();
244  interest->setNonce(m_rand->GetValue(0, std::numeric_limits<uint32_t>::max()));
245  interest->setSubscription(m_subscription);
246  if (m_subscription == 0) {
247  nameWithSequence->appendSequenceNumber(seq); //Required for demand-response scheme [payload interest] (also usefule for ndn::AppDelayTracer)
248  interest->setPayload(payload, m_virtualPayloadSize); //add payload to interest
249  }
250  interest->setName(*nameWithSequence);
251  time::milliseconds interestLifeTime(m_interestLifeTime.GetMilliSeconds());
252  interest->setInterestLifetime(interestLifeTime);
253 
254  NS_LOG_INFO("node(" << GetNode()->GetId() << ") > sending Interest: " << interest->getName() /*m_interestName*/ << " with Payload = " << interest->getPayloadLength() << "bytes");
255 
256  WillSendOutInterest(seq);
257 
258  m_transmittedInterests(interest, this, m_face);
259  m_face->onReceiveInterest(*interest);
260 
261  // Callback for sent payload interests
262  m_sentInterest(GetNode()->GetId(), interest);
263 
264  ScheduleNextPacket();
265

275 {
276  if (!m_active)
277  return;

279  App::OnData(data); // tracing inside

281  NS_LOG_FUNCTION(this << data);

283  // This could be a problem......

285 
286  NS_LOG_INFO("node(" << GetNode()->GetId() << ") < Received DATA for " << /*m_interestName*/ data->getName() << " TIME: " << Simulator::Now());
287 
288  // Callback for received subscription data
289  m_receivedData(GetNode()->GetId(), data);
290 
291  int hopCount = 0;
292  auto ns3PacketTag = data->getTag<Ns3PacketTag>();
293  if (ns3PacketTag != nullptr) { // e.g., packet came from local node's cache
294  FwHopCountTag hopCountTag;
295  if (ns3PacketTag->getPacket()->PeekPacketTag(hopCountTag)) {
296  hopCount = hopCountTag.Get();
297  NS_LOG_DEBUG("Hop count: " << hopCount);
298  }
299  }
300 
301  //Enable trace file for Interests with sequence number (subscription = 0)
302  if (m_subscription == 0) {
303 
304  // This could be a problem......
305  uint32_t seq = data->getName().at(-1).toSequenceNumber();
306 
307  SeqTimeoutsContainer::iterator entry = m_seqLastDelay.find(seq);
308  if (entry != m_seqLastDelay.end()) {
309  m_lastRetransmittedInterestDataDelay(this, seq, Simulator::Now() - entry->time, hopCount);
310  }
311 
312  entry = m_seqFullDelay.find(seq);
313  if (entry != m_seqFullDelay.end()) {
314  m_firstInterestDataDelay(this, seq, Simulator::Now() - entry->time, m_seqRetxCounts[seq], hopCount);
315  }
316 
317  m_seqRetxCounts.erase(seq);
318  m_seqFullDelay.erase(seq);
319  m_seqLastDelay.erase(seq);
320 
321  m_seqTimeouts.erase(seq);
322  m_retxSeqs.erase(seq);
323
