111  std::string message(int value) const {
112  switch(value) {
113  case error::security:
114  return "Security policy error";
115  case error::socket:
116  return "Socket component error";
117  case error::invalid_state:
118  return "Invalid state";
119  case error::invalid_tls_context:

121  case error::tls_handshake_timeout:
122  return "TLS handshake timed out";
123  case error::pass_through:
124  return "Pass through from socket policy";
125  case error::missing_tls_init_handler:
126  return "Required tls_init handler not present.";
127  case error::tls_handshake_failed:
128  return "TLS handshake failed";
129  default:
130  return "Unknown";
131  }
132  }
