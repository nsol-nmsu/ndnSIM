150 bool
151 Interest::matchesName(const Name& name) const
152 {
153  if (name.size() < m_name.size())
154  return false;
155 
156  if (!m_name.isPrefixOf(name))
157  return false;
158 
159  if (getMinSuffixComponents() >= 0 &&
160  // name must include implicit digest
161  !(name.size() - m_name.size() >= static_cast<size_t>(getMinSuffixComponents())))
162  return false;
163 
164  if (getMaxSuffixComponents() >= 0 &&
165  // name must include implicit digest
166  !(name.size() - m_name.size() <= static_cast<size_t>(getMaxSuffixComponents())))
167  return false;
168 
169  if (!getExclude().empty() &&
170  name.size() > m_name.size() &&
171  getExclude().isExcluded(name[m_name.size()]))
172  return false;
173 
174  return true;
175 }

177 bool
178 Interest::matchesData(const Data& data) const
179 {
180  size_t interestNameLength = m_name.size();
181  const Name& dataName = data.getName();
182  size_t fullNameLength = dataName.size() + 1;
183 
184  // check MinSuffixComponents
185  bool hasMinSuffixComponents = getMinSuffixComponents() >= 0;
186  size_t minSuffixComponents = hasMinSuffixComponents ?
187  static_cast<size_t>(getMinSuffixComponents()) : 0;
188  if (!(interestNameLength + minSuffixComponents <= fullNameLength))
189  return false;
190 
191  // check MaxSuffixComponents
192  bool hasMaxSuffixComponents = getMaxSuffixComponents() >= 0;
193  if (hasMaxSuffixComponents &&
194  !(interestNameLength + getMaxSuffixComponents() >= fullNameLength))
195  return false;
196 
197  // check prefix
198  if (interestNameLength == fullNameLength) {
199  if (m_name.get(-1).isImplicitSha256Digest()) {
200  if (m_name != data.getFullName())
201  return false;
202  }
203  else {
204  // Interest Name is same length as Data full Name, but last component isn't digest
205  // so there's no possibility of matching
206  return false;
207  }
208  }
209  else {
210  // Interest Name is a strict prefix of Data full Name
211  if (!m_name.isPrefixOf(dataName))
212  return false;
213  }
214 
215  // check Exclude
216  // Exclude won't be violated if Interest Name is same as Data full Name
217  if (!getExclude().empty() && fullNameLength > interestNameLength) {
218  if (interestNameLength == fullNameLength - 1) {
219  // component to exclude is the digest
220  if (getExclude().isExcluded(data.getFullName().get(interestNameLength)))
221  return false;
222  // There's opportunity to inspect the Exclude filter and determine whether
223  // the digest would make a difference.
224  // eg. "<NameComponent>AA</NameComponent><Any/>" doesn't exclude any digest -
225  // fullName not needed;
226  // "<Any/><NameComponent>AA</NameComponent>" and
227  // "<Any/><ImplicitSha256DigestComponent>ffffffffffffffffffffffffffffffff
228  // </ImplicitSha256DigestComponent>"
229  // excludes all digests - fullName not needed;
230  // "<Any/><ImplicitSha256DigestComponent>80000000000000000000000000000000
231  // </ImplicitSha256DigestComponent>"
232  // excludes some digests - fullName required
233  // But Interests that contain the exact Data Name before digest and also
234  // contain Exclude filter is too rare to optimize for, so we request
235  // fullName no mater what's in the Exclude filter.
236  }
237  else {
238  // component to exclude is not the digest
239  if (getExclude().isExcluded(dataName.get(interestNameLength)))
240  return false;
241  }
242  }
243 
244  // check PublisherPublicKeyLocator
245  const KeyLocator& publisherPublicKeyLocator = this->getPublisherPublicKeyLocator();
246  if (!publisherPublicKeyLocator.empty()) {
247  const Signature& signature = data.getSignature();
248  const Block& signatureInfo = signature.getInfo();
249  Block::element_const_iterator it = signatureInfo.find(tlv::KeyLocator);
250  if (it == signatureInfo.elements_end()) {
251  return false;
252  }
253  if (publisherPublicKeyLocator.wireEncode() != *it) {
254  return false;
255  }
256  }
257 
258  return true;
259 }

261 template<encoding::Tag TAG>
262 size_t
263 Interest::wireEncode(EncodingImpl<TAG>& encoder) const
264 {
265  size_t totalLength = 0;
266 
267  // Interest ::= INTEREST-TYPE TLV-LENGTH
268  // Name
269  // Selectors?
270  // Nonce
271  // InterestLifetime?
272  // Link?
273  // SelectedDelegation?
274 
275  // (reverse encoding)
276 
277  // Payload
278  if(getPayloadLength() != 0) {
279  totalLength += encoder.prependBlock(m_payload);
280  }
281 
282  // Subscription
283  if (getSubscription() >= 0) {
284  totalLength += prependNonNegativeIntegerBlock(encoder,
285  tlv::Subscription,
286  getSubscription());
287  }
288 
289  if (hasLink()) {
290  if (hasSelectedDelegation()) {
291  totalLength += prependNonNegativeIntegerBlock(encoder,
292  tlv::SelectedDelegation,
293  m_selectedDelegationIndex);
294  }
295  totalLength += encoder.prependBlock(m_link);
296  }
297  else {
298  BOOST_ASSERT(!hasSelectedDelegation());
299  }
300 
301  // InterestLifetime
302  if (getInterestLifetime() >= time::milliseconds::zero() &&
303  getInterestLifetime() != DEFAULT_INTEREST_LIFETIME)
304  {
305  totalLength += prependNonNegativeIntegerBlock(encoder,
306  tlv::InterestLifetime,
307  getInterestLifetime().count());
308  }
309 
310  // Nonce
311  getNonce(); // to ensure that Nonce is properly set
312  totalLength += encoder.prependBlock(m_nonce);
313 
314  // Selectors
315  if (hasSelectors())
316  {
317  totalLength += getSelectors().wireEncode(encoder);
318  }
319 
320  // Name
321  totalLength += getName().wireEncode(encoder);
322 
323  totalLength += encoder.prependVarNumber(totalLength);
324  totalLength += encoder.prependVarNumber(tlv::Interest);
325  return totalLength;
326 }

352 void
353 Interest::wireDecode(const Block& wire)
354 {
355  m_wire = wire;
356  m_wire.parse();
357 
358  // Interest ::= INTEREST-TYPE TLV-LENGTH
359  // Name
360  // Selectors?
361  // Nonce
362  // InterestLifetime?
363  // Link?
364  // SelectedDelegation?
365 
366  if (m_wire.type() != tlv::Interest)
367  BOOST_THROW_EXCEPTION(Error("Unexpected TLV number when decoding Interest"));
368 
369  // Name
370  m_name.wireDecode(m_wire.get(tlv::Name));
371 
372  // Selectors
373  Block::element_const_iterator val = m_wire.find(tlv::Selectors);
374  if (val != m_wire.elements_end())
375  {
376  m_selectors.wireDecode(*val);
377  }
378  else
379  m_selectors = Selectors();
380 
381  // Nonce
382  m_nonce = m_wire.get(tlv::Nonce);
383 
384  // InterestLifetime
385  val = m_wire.find(tlv::InterestLifetime);
386  if (val != m_wire.elements_end())
387  {
388  m_interestLifetime = time::milliseconds(readNonNegativeInteger(*val));
389  }
390  else
391  {
392  m_interestLifetime = DEFAULT_INTEREST_LIFETIME;
393  }
394 
395  // Link object
396  val = m_wire.find(tlv::Data);
397  if (val != m_wire.elements_end())
398  {
399  m_link = (*val);
400  }
401 
402  // SelectedDelegation
403  val = m_wire.find(tlv::SelectedDelegation);
404  if (val != m_wire.elements_end()) {
405  if (!this->hasLink()) {
406  BOOST_THROW_EXCEPTION(Error("Interest contains selectedDelegation, but no LINK object"));
407  }
408  uint64_t selectedDelegation = readNonNegativeInteger(*val);
409  if (selectedDelegation < uint64_t(Link::countDelegationsFromWire(m_link))) {
410  m_selectedDelegationIndex = static_cast<size_t>(selectedDelegation);
411  }
412  else {
413  BOOST_THROW_EXCEPTION(Error("Invalid selected delegation index when decoding Interest"));
414  }
415  }
416  else {
417  m_selectedDelegationIndex = INVALID_SELECTED_DELEGATION_INDEX;
418  }
419 
420  // Subscription
421  val = m_wire.find(tlv::Subscription);
422  if (val != m_wire.elements_end()) {
423  m_subscribe = readNonNegativeInteger(*val);
424  }
425  else {
426  m_subscribe = 0;
427  }
428 
429  // Payload
430  val = m_wire.find(tlv::Payload);
431  if (val != m_wire.elements_end()){
432  m_payload = *val;
433  }
434 
435 }
