166 
167  m_buffer.appendByteArray(payload.value(), payload.value_size());
168 
169  BOOST_CHECK_NO_THROW(data.getName()[-1].toSegment());

172  return;

174 
175  // wrap the Face Statuses in a single Content TLV for easy parsing
176  m_buffer.prependVarNumber(m_buffer.size());
177  m_buffer.prependVarNumber(tlv::Content);
178 
179  ndn::Block parser(m_buffer.buf(), m_buffer.size());
180  parser.parse();
181 
182  BOOST_REQUIRE_EQUAL(parser.elements_size(), m_referenceFaces.size());
183

185  for (Block::element_const_iterator i = parser.elements_begin();
