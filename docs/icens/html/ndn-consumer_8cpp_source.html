115 void
116 Consumer::CheckRetxTimeout()
117 {
118  Time now = Simulator::Now();
119 
120  Time rto = m_rtt->RetransmitTimeout();
121  // NS_LOG_DEBUG ("Current RTO: " << rto.ToDouble (Time::S) << "s");
122 
123  while (!m_seqTimeouts.empty()) {
124  SeqTimeoutsContainer::index<i_timestamp>::type::iterator entry =
125  m_seqTimeouts.get<i_timestamp>().begin();
126  if (entry->time + rto <= now) // timeout expired?
127  {
128  uint32_t seqNo = entry->seq;
129  m_seqTimeouts.get<i_timestamp>().erase(entry);
130  OnTimeout(seqNo);
131  }
132  else
133  break; // nothing else to do. All later packets need not be retransmitted
134  }
135 
136  m_retxEvent = Simulator::Schedule(m_retxTimer, &Consumer::CheckRetxTimeout, this);
137 }

169  NS_LOG_FUNCTION_NOARGS();

172

175  m_retxSeqs.erase(m_retxSeqs.begin());
176  break;
177  }
178 
179  if (seq == std::numeric_limits<uint32_t>::max()) {
180  if (m_seqMax != std::numeric_limits<uint32_t>::max()) {
181  if (m_seq >= m_seqMax) {
182  return; // we are totally done
183  }
184  }
185 
186  seq = m_seq++;
187  }
188 
189  //
190  shared_ptr<Name> nameWithSequence = make_shared<Name>(m_interestName);
191  nameWithSequence->appendSequenceNumber(seq);
192  //
193 
194  // shared_ptr<Interest> interest = make_shared<Interest> ();
195  shared_ptr<Interest> interest = make_shared<Interest>();
196  interest->setNonce(m_rand->GetValue(0, std::numeric_limits<uint32_t>::max()));
197  interest->setName(*nameWithSequence);
198  time::milliseconds interestLifeTime(m_interestLifeTime.GetMilliSeconds());
199  interest->setInterestLifetime(interestLifeTime);
200 
201  // NS_LOG_INFO ("Requesting Interest: \n" << *interest);
202  NS_LOG_INFO("> Interest for " << seq);

216 void
217 Consumer::OnData(shared_ptr<const Data> data)
218 {
219  if (!m_active)
220  return;
221 
222  App::OnData(data); // tracing inside
223 
224  NS_LOG_FUNCTION(this << data);
225 
226  // NS_LOG_INFO ("Received content object: " << boost::cref(*data));
227 
228  // This could be a problem......
229  uint32_t seq = data->getName().at(-1).toSequenceNumber();
230  NS_LOG_INFO("< DATA for " << seq);
231 
232  int hopCount = 0;
233  auto ns3PacketTag = data->getTag<Ns3PacketTag>();
234  if (ns3PacketTag != nullptr) { // e.g., packet came from local node's cache
235  FwHopCountTag hopCountTag;
236  if (ns3PacketTag->getPacket()->PeekPacketTag(hopCountTag)) {
237  hopCount = hopCountTag.Get();
238  NS_LOG_DEBUG("Hop count: " << hopCount);
239  }
240  }
241 
242  SeqTimeoutsContainer::iterator entry = m_seqLastDelay.find(seq);
243  if (entry != m_seqLastDelay.end()) {
244  m_lastRetransmittedInterestDataDelay(this, seq, Simulator::Now() - entry->time, hopCount);
245  }
246 
247  entry = m_seqFullDelay.find(seq);
248  if (entry != m_seqFullDelay.end()) {
249  m_firstInterestDataDelay(this, seq, Simulator::Now() - entry->time, m_seqRetxCounts[seq], hopCount);
250  }
251 
252  m_seqRetxCounts.erase(seq);
253  m_seqFullDelay.erase(seq);
254  m_seqLastDelay.erase(seq);
255 
256  m_seqTimeouts.erase(seq);
257  m_retxSeqs.erase(seq);
258 
259  m_rtt->AckSeq(SequenceNumber32(seq));
260 }
