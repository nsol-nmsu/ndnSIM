265 {
266  if (begin == end)
267  return false;
268 
269  uint8_t firstOctet = *begin;
270  ++begin;
271  if (firstOctet < 253)
272  {
273  number = firstOctet;
274  }
275  else if (firstOctet == 253)
276  {
277  if (end - begin < 2)
278  return false;
279 
280  uint16_t value = *reinterpret_cast<const uint16_t*>(&*begin);
281  begin += 2;
282  number = be16toh(value);
283  }
284  else if (firstOctet == 254)
285  {
286  if (end - begin < 4)

295  if (end - begin < 8)

299  begin += 8;

301  number = be64toh(value);
302  }
303 
304  return true;
