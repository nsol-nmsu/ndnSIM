217 {
218  Data data("ndn:/A");
219  data.setFreshnessPeriod(time::seconds(10));
220  static const uint8_t someData[] = "someData";
221  data.setContent(someData, sizeof(someData));
222  data.setSignature(SignatureSha256WithRsa());
223  data.setCachingPolicy(nfd::LocalControlHeader::CachingPolicy::NO_CACHE);
224 
225  BOOST_CHECK(!data.hasWire());
226 
227  Block headerBlock =
228  data.getLocalControlHeader().wireEncode(data, nfd::LocalControlHeader::ENCODE_CACHING_POLICY);
229 
230  BOOST_CHECK(data.hasWire());
231  BOOST_CHECK(headerBlock.hasWire());
232 
233  BOOST_CHECK_NE(headerBlock.wire(), data.wireEncode().wire());
234  BOOST_CHECK_NE(headerBlock.size(), data.wireEncode().size());
235  BOOST_CHECK_EQUAL(headerBlock.size(), 6);
236 
237  BOOST_CHECK_EQUAL_COLLECTIONS(DataWithLocalControlHeader,
238  DataWithLocalControlHeader + 6,
239  headerBlock.begin(), headerBlock.end());
240 
241  data.setFreshnessPeriod(time::seconds(1000));
242 
243  Block updatedHeaderBlock = data.getLocalControlHeader()
244  .wireEncode(data, nfd::LocalControlHeader::ENCODE_CACHING_POLICY);
245  BOOST_CHECK_EQUAL(updatedHeaderBlock.size(), 6);
246 
247  // only length should have changed
248  BOOST_CHECK_EQUAL_COLLECTIONS(updatedHeaderBlock.begin() + 2, updatedHeaderBlock.end(),
249  headerBlock.begin() + 2, headerBlock.end());
250 
251  // adding IncomingFaceId
252  data.setIncomingFaceId(10);
253  updatedHeaderBlock =
254  data.getLocalControlHeader().wireEncode(data, nfd::LocalControlHeader::ENCODE_INCOMING_FACE_ID
255  | nfd::LocalControlHeader::ENCODE_CACHING_POLICY);
256  BOOST_CHECK_EQUAL(updatedHeaderBlock.size(), 9);
257 
258  // masking CachingPolicy
259  updatedHeaderBlock = data.getLocalControlHeader()

261  BOOST_CHECK_EQUAL(updatedHeaderBlock.size(), 5);

265  .wireEncode(data, nfd::LocalControlHeader::ENCODE_NONE),
266  nfd::LocalControlHeader::Error);
