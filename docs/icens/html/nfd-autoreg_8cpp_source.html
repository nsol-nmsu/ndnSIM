124  void
125  registerPrefixesForFace(uint64_t faceId,
126  const std::vector<ndn::Name>& prefixes)
127  {

137  .setCost(m_cost)
138  .setExpirationPeriod(time::milliseconds::max()),
139  bind(&AutoregServer::onRegisterCommandSuccess, this, faceId, *prefix),

144  void
145  registerPrefixesIfNeeded(uint64_t faceId, const FaceUri& uri, FacePersistency facePersistency)
146  {
147  if (hasAllowedSchema(uri)) {
148  boost::system::error_code ec;
149  boost::asio::ip::address address = boost::asio::ip::address::from_string(uri.getHost(), ec);
150 
151  if (!address.is_multicast()) {
152  // register all-face prefixes
153  registerPrefixesForFace(faceId, m_allFacesPrefixes);
154 
155  // register autoreg prefixes if new face is on-demand and not blacklisted and whitelisted
156  if (facePersistency == FACE_PERSISTENCY_ON_DEMAND &&
157  !isBlacklisted(address) && isWhitelisted(address)) {
158  registerPrefixesForFace(faceId, m_autoregPrefixes);
159  }
160  }
161  }

247  void
248  fetchFaceStatusSegments(const Data& data, const shared_ptr<ndn::OBufferStream>& buffer)
249  {
250  buffer->write(reinterpret_cast<const char*>(data.getContent().value()),
251  data.getContent().value_size());
252 
253  uint64_t currentSegment = data.getName().get(-1).toSegment();
254 
255  const name::Component& finalBlockId = data.getMetaInfo().getFinalBlockId();
256  if (finalBlockId.empty() || finalBlockId.toSegment() > currentSegment) {
257  m_face.expressInterest(data.getName().getPrefix(-1).appendSegment(currentSegment + 1),
258  bind(&AutoregServer::fetchFaceStatusSegments, this, _2, buffer),
259  ndn::OnTimeout());
260  }
261  else {
262  return processFaceStatusDataset(buffer);
263  }
264  }

280  void
281  processFaceStatusDataset(const shared_ptr<ndn::OBufferStream>& buffer)
282  {
283  ndn::ConstBufferPtr buf = buffer->buf();
284  std::vector<uint64_t> multicastFaces;
285 
286  size_t offset = 0;
287  while (offset < buf->size()) {
288  bool isOk = false;
289  Block block;
290  std::tie(isOk, block) = Block::fromBuffer(buf, offset);
291  if (!isOk) {
292  std::cerr << "ERROR: cannot decode FaceStatus TLV" << std::endl;
293  break;
294  }
295 
296  offset += block.size();
297 
298  nfd::FaceStatus faceStatus(block);
299  registerPrefixesIfNeeded(faceStatus.getFaceId(), FaceUri(faceStatus.getRemoteUri()),
300  faceStatus.getFacePersistency());
301  }
