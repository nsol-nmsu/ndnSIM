88  err_str_pair negotiate_extensions(request_type const & req) {
89  err_str_pair ret;
90 
91  // Respect blanket disabling of all extensions and don't even parse
92  // the extension header
93  if (!config::enable_extensions) {
94  ret.first = make_error_code(error::extensions_disabled);
95  return ret;
96  }
97 
98  http::parameter_list p;
99 
100  bool error = req.get_header_as_plist("Sec-WebSocket-Extensions",p);
101 
102  if (error) {
103  ret.first = make_error_code(error::extension_parse_error);
104  return ret;
105  }
106 
107  // If there are no extensions parsed then we are done!
108  if (p.size() == 0) {
109  return ret;
110  }
111 
112  http::parameter_list::const_iterator it;
113 
114  if (m_permessage_deflate.is_implemented()) {
115  err_str_pair neg_ret;
116  for (it = p.begin(); it != p.end(); ++it) {
117  // look through each extension, if the key is permessage-deflate
118  if (it->first == "permessage-deflate") {
119  neg_ret = m_permessage_deflate.negotiate(it->second);
120 
121  if (neg_ret.first) {
122  // Figure out if this is an error that should halt all
123  // extension negotiations or simply cause negotiation of
124  // this specific extension to fail.
125  //std::cout << "permessage-compress negotiation failed: "
126  // << neg_ret.first.message() << std::endl;
127  } else {
128  // Note: this list will need commas if WebSocket++ ever
129  // supports more than one extension
130  ret.second += neg_ret.second;
131  continue;
132  }
133  }
134  }
135  }
136 
137  return ret;
138  }

140  lib::error_code validate_handshake(request_type const & r) const {
141  if (r.get_method() != "GET") {

154  return make_error_code(error::missing_required_header);
155  }

157  return lib::error_code();

164  lib::error_code process_handshake(request_type const & request,
165  std::string const & subprotocol, response_type & response) const
166  {
167  std::string server_key = request.get_header("Sec-WebSocket-Key");
168 
169  lib::error_code ec = process_handshake_key(server_key);
170 
171  if (ec) {
172  return ec;
173  }
174 
175  response.replace_header("Sec-WebSocket-Accept",server_key);
176  response.append_header("Upgrade",constants::upgrade_token);
177  response.append_header("Connection",constants::connection_token);
178 
179  if (!subprotocol.empty()) {
180  response.replace_header("Sec-WebSocket-Protocol",subprotocol);
181  }
182 
183  return lib::error_code();
184  }

192  lib::error_code client_handshake_request(request_type & req, uri_ptr
193  uri, std::vector<std::string> const & subprotocols) const
194  {
195  req.set_method("GET");
196  req.set_uri(uri->get_resource());
197  req.set_version("HTTP/1.1");
198 
199  req.append_header("Upgrade","websocket");
200  req.append_header("Connection","Upgrade");
201  req.replace_header("Sec-WebSocket-Version","13");
202  req.replace_header("Host",uri->get_host_port());
203 
204  if (!subprotocols.empty()) {
205  std::ostringstream result;
206  std::vector<std::string>::const_iterator it = subprotocols.begin();
207  result << *it++;
208  while (it != subprotocols.end()) {
209  result << ", " << *it++;
210  }
211 
212  req.replace_header("Sec-WebSocket-Protocol",result.str());
213  }
214 
215  // Generate handshake key
216  frame::uint32_converter conv;
217  unsigned char raw_key[16];
218 
219  for (int i = 0; i < 4; i++) {
220  conv.i = m_rng();
221  std::copy(conv.c,conv.c+4,&raw_key[i*4]);
222  }
223 
224  req.replace_header("Sec-WebSocket-Key",base64_encode(raw_key, 16));
225 
226  return lib::error_code();
227  }

235  lib::error_code validate_server_handshake_response(request_type const & req,
236  response_type& res) const
237  {
238  // A valid response has an HTTP 101 switching protocols code
239  if (res.get_status_code() != http::status_code::switching_protocols) {
240  return error::make_error_code(error::invalid_http_status);
241  }
242 
243  // And the upgrade token in an upgrade header
244  std::string const & upgrade_header = res.get_header("Upgrade");
245  if (utility::ci_find_substr(upgrade_header, constants::upgrade_token,
246  sizeof(constants::upgrade_token)-1) == upgrade_header.end())
247  {
248  return error::make_error_code(error::missing_required_header);
249  }
250 
251  // And the websocket token in the connection header
252  std::string const & con_header = res.get_header("Connection");
253  if (utility::ci_find_substr(con_header, constants::connection_token,
254  sizeof(constants::connection_token)-1) == con_header.end())
255  {
256  return error::make_error_code(error::missing_required_header);
257  }
258 
259  // And has a valid Sec-WebSocket-Accept value
260  std::string key = req.get_header("Sec-WebSocket-Key");
261  lib::error_code ec = process_handshake_key(key);
262 
263  if (ec || key != res.get_header("Sec-WebSocket-Accept")) {
264  return error::make_error_code(error::missing_required_header);
265  }
266 
267  return lib::error_code();
268  }

328  size_t consume(uint8_t * buf, size_t len, lib::error_code & ec) {
329  size_t p = 0;
330 
331  ec = lib::error_code();
332 
333  //std::cout << "consume: " << utility::to_hex(buf,len) << std::endl;
334 
335  // Loop while we don't have a message ready and we still have bytes
336  // left to process.
337  while (m_state != READY && m_state != FATAL_ERROR &&
338  (p < len || m_bytes_needed == 0))
339  {
340  if (m_state == HEADER_BASIC) {
341  p += this->copy_basic_header_bytes(buf+p,len-p);
342 
343  if (m_bytes_needed > 0) {
344  continue;
345  }
346 
347  ec = this->validate_incoming_basic_header(
348  m_basic_header, base::m_server, !m_data_msg.msg_ptr
349  );
350  if (ec) {break;}
351 
352  // extract full header size and adjust consume state accordingly
353  m_state = HEADER_EXTENDED;
354  m_cursor = 0;
355  m_bytes_needed = frame::get_header_len(m_basic_header) -
356  frame::BASIC_HEADER_LENGTH;
357  } else if (m_state == HEADER_EXTENDED) {
358  p += this->copy_extended_header_bytes(buf+p,len-p);
359 
360  if (m_bytes_needed > 0) {
361  continue;
362  }
363 
364  ec = validate_incoming_extended_header(m_basic_header,m_extended_header);
365  if (ec){break;}
366 
367  m_state = APPLICATION;
368  m_bytes_needed = static_cast<size_t>(get_payload_size(m_basic_header,m_extended_header));
369 
370  // check if this frame is the start of a new message and set up
371  // the appropriate message metadata.
372  frame::opcode::value op = frame::get_opcode(m_basic_header);
373 
374  // TODO: get_message failure conditions
375 
376  if (frame::opcode::is_control(op)) {
377  m_control_msg = msg_metadata(
378  m_msg_manager->get_message(op,m_bytes_needed),
379  frame::get_masking_key(m_basic_header,m_extended_header)
380  );
381 
382  m_current_msg = &m_control_msg;
383  } else {
384  if (!m_data_msg.msg_ptr) {
385  if (m_bytes_needed > base::m_max_message_size) {
386  ec = make_error_code(error::message_too_big);
387  break;
388  }
389 
390  m_data_msg = msg_metadata(
391  m_msg_manager->get_message(op,m_bytes_needed),
392  frame::get_masking_key(m_basic_header,m_extended_header)
393  );
394  } else {
395  // Fetch the underlying payload buffer from the data message we
396  // are writing into.
397  std::string & out = m_data_msg.msg_ptr->get_raw_payload();
398 
399  if (out.size() + m_bytes_needed > base::m_max_message_size) {
400  ec = make_error_code(error::message_too_big);
401  break;
402  }
403 
404  // Each frame starts a new masking key. All other state
405  // remains between frames.
406  m_data_msg.prepared_key = prepare_masking_key(
407  frame::get_masking_key(
408  m_basic_header,
409  m_extended_header
410  )
411  );
412 
413  out.reserve(out.size() + m_bytes_needed);
414  }
415  m_current_msg = &m_data_msg;
416  }
417  } else if (m_state == EXTENSION) {
418  m_state = APPLICATION;
419  } else if (m_state == APPLICATION) {
420  size_t bytes_to_process = (std::min)(m_bytes_needed,len-p);
421 
422  if (bytes_to_process > 0) {
423  p += this->process_payload_bytes(buf+p,bytes_to_process,ec);
424 
425  if (ec) {break;}
426  }
427 
428  if (m_bytes_needed > 0) {
429  continue;
430  }
431 
432  // If this was the last frame in the message set the ready flag.
433  // Otherwise, reset processor state to read additional frames.
434  if (frame::get_fin(m_basic_header)) {
435  // ensure that text messages end on a valid UTF8 code point
436  if (frame::get_opcode(m_basic_header) == frame::opcode::TEXT) {
437  if (!m_current_msg->validator.complete()) {
438  ec = make_error_code(error::invalid_utf8);
439  break;
440  }
441  }
442 
443  m_state = READY;
444  } else {
445  this->reset_headers();
446  }
447  } else {
448  // shouldn't be here
449  ec = make_error_code(error::general);
450  return 0;
451  }
452  }
453 
454  return p;
455  }

527  virtual lib::error_code prepare_data_frame(message_ptr in, message_ptr out)
528  {
529  if (!in || !out) {
530  return make_error_code(error::invalid_arguments);
531  }
532 
533  frame::opcode::value op = in->get_opcode();
534 
535  // validate opcode: only regular data frames
536  if (frame::opcode::is_control(op)) {
537  return make_error_code(error::invalid_opcode);
538  }
539 
540  std::string& i = in->get_raw_payload();
541  std::string& o = out->get_raw_payload();
542 
543  // validate payload utf8
544  if (op == frame::opcode::TEXT && !utf8_validator::validate(i)) {
545  return make_error_code(error::invalid_payload);
546  }
547 
548  frame::masking_key_type key;
549  bool masked = !base::m_server;
550  bool compressed = m_permessage_deflate.is_enabled()
551  && in->get_compressed();
552  bool fin = in->get_fin();
553 
554  // generate header
555  frame::basic_header h(op,i.size(),fin,masked,compressed);
556 
557  if (masked) {
558  // Generate masking key.
559  key.i = m_rng();
560 
561  frame::extended_header e(i.size(),key.i);
562  out->set_header(frame::prepare_header(h,e));
563  } else {
564  frame::extended_header e(i.size());
565  out->set_header(frame::prepare_header(h,e));
566  }
567 
568  // prepare payload
569  if (compressed) {
570  // compress and store in o after header.
571  m_permessage_deflate.compress(i,o);
572 
573  // mask in place if necessary
574  if (masked) {
575  this->masked_copy(o,o,key);
576  }
577  } else {
578  // no compression, just copy data into the output buffer
579  o.resize(i.size());
580 
581  // if we are masked, have the masking function write to the output
582  // buffer directly to avoid another copy. If not masked, copy
583  // directly without masking.
584  if (masked) {
585  this->masked_copy(i,o,key);
586  } else {
587  std::copy(i.begin(),i.end(),o.begin());
588  }
589  }
590 
591  out->set_prepared(true);
592  out->set_opcode(op);
593 
594  return lib::error_code();
595  }

606  virtual lib::error_code prepare_close(close::status::value code,
607  std::string const & reason, message_ptr out) const
608  {
609  if (close::status::reserved(code)) {
610  return make_error_code(error::reserved_close_code);
611  }
612 
613  if (close::status::invalid(code) && code != close::status::no_status) {
614  return make_error_code(error::invalid_close_code);
615  }
616 
617  if (code == close::status::no_status && reason.size() > 0) {
618  return make_error_code(error::reason_requires_code);
619  }
620 
621  if (reason.size() > frame:: limits::payload_size_basic-2) {
622  return make_error_code(error::control_too_big);
623  }
624 
625  std::string payload;
626 
627  if (code != close::status::no_status) {
628  close::code_converter val;
629  val.i = htons(code);
630 
631  payload.resize(reason.size()+2);
632 
633  payload[0] = val.c[0];
634  payload[1] = val.c[1];
635 
636  std::copy(reason.begin(),reason.end(),payload.begin()+2);
637  }
638 
639  return this->prepare_control(frame::opcode::CLOSE,payload,out);
640  }

654  size_t copy_basic_header_bytes(uint8_t const * buf, size_t len) {
655  if (len == 0 || m_bytes_needed == 0) {
656  return 0;
657  }
658 
659  if (len > 1) {
660  // have at least two bytes
661  if (m_bytes_needed == 2) {
662  m_basic_header.b0 = buf[0];
663  m_basic_header.b1 = buf[1];
664  m_bytes_needed -= 2;
665  return 2;
666  } else {
667  m_basic_header.b1 = buf[0];
668  m_bytes_needed--;
669  return 1;
670  }
671  } else {
672  // have exactly one byte
673  if (m_bytes_needed == 2) {
674  m_basic_header.b0 = buf[0];
675  m_bytes_needed--;
676  return 1;
677  } else {
678  m_basic_header.b1 = buf[0];
679  m_bytes_needed--;
680  return 1;
681  }
682  }
683  }

686  size_t copy_extended_header_bytes(uint8_t const * buf, size_t len) {

709  size_t process_payload_bytes(uint8_t * buf, size_t len, lib::error_code& ec)
710  {
711  // unmask if masked
712  if (frame::get_masked(m_basic_header)) {
713  #ifdef WEBSOCKETPP_STRICT_MASKING
714  m_current_msg->prepared_key = frame::byte_mask_circ(
715  buf,
716  len,
717  m_current_msg->prepared_key
718  );
719  #else
720  m_current_msg->prepared_key = frame::word_mask_circ(
721  buf,
722  len,
723  m_current_msg->prepared_key
724  );
725  #endif
726  }
727 
728  std::string & out = m_current_msg->msg_ptr->get_raw_payload();
729  size_t offset = out.size();
730 
731  // decompress message if needed.
732  if (m_permessage_deflate.is_enabled()
733  && frame::get_rsv1(m_basic_header))
734  {
735  // Decompress current buffer into the message buffer
736  m_permessage_deflate.decompress(buf,len,out);
737 
738  // get the length of the newly uncompressed output
739  offset = out.size() - offset;
740  } else {
741  // No compression, straight copy
742  out.append(reinterpret_cast<char *>(buf),len);
743  }
744 
745  // validate unmasked, decompressed values
746  if (m_current_msg->msg_ptr->get_opcode() == frame::opcode::TEXT) {
747  if (!m_current_msg->validator.decode(out.begin()+offset,out.end())) {
748  ec = make_error_code(error::invalid_utf8);
749  return 0;
750  }
751  }
752 
753  m_bytes_needed -= len;
754 
755  return len;
756  }

768  lib::error_code validate_incoming_basic_header(frame::basic_header const & h,
769  bool is_server, bool new_msg) const
770  {
771  frame::opcode::value op = frame::get_opcode(h);
772 
773  // Check control frame size limit
774  if (frame::opcode::is_control(op) &&
775  frame::get_basic_size(h) > frame::limits::payload_size_basic)
776  {
777  return make_error_code(error::control_too_big);
778  }
779 
780  // Check that RSV bits are clear
781  // The only RSV bits allowed are rsv1 if the permessage_compress
782  // extension is enabled for this connection and the message is not
783  // a control message.
784  //
785  // TODO: unit tests for this
786  if (frame::get_rsv1(h) && (!m_permessage_deflate.is_enabled()
787  || frame::opcode::is_control(op)))
788  {
789  return make_error_code(error::invalid_rsv_bit);
790  }
791

793  return make_error_code(error::invalid_rsv_bit);
794  }
795 
796  // Check for reserved opcodes
797  if (frame::opcode::reserved(op)) {
798  return make_error_code(error::invalid_opcode);
799  }
800 
801  // Check for invalid opcodes
802  // TODO: unit tests for this?
803  if (frame::opcode::invalid(op)) {
804  return make_error_code(error::invalid_opcode);
805  }
806 
807  // Check for fragmented control message
808  if (frame::opcode::is_control(op) && !frame::get_fin(h)) {
809  return make_error_code(error::fragmented_control);
810  }
811 
812  // Check for continuation without an active message
813  if (new_msg && op == frame::opcode::CONTINUATION) {
814  return make_error_code(error::invalid_continuation);
815  }
816 
817  // Check for new data frame when expecting continuation
818  if (!new_msg && !frame::opcode::is_control(op) &&
819  op != frame::opcode::CONTINUATION)
820  {
821  return make_error_code(error::invalid_continuation);
822  }
823 
824  // Servers should reject any unmasked frames from clients.
825  // Clients should reject any masked frames from servers.
826  if (is_server && !frame::get_masked(h)) {
827  return make_error_code(error::masking_required);
828  } else if (!is_server && frame::get_masked(h)) {
829  return make_error_code(error::masking_forbidden);
830  }
831 
832  return lib::error_code();
833  }

850  uint64_t payload_size = frame::get_payload_size(h,e);
851

854  payload_size <= frame::limits::payload_size_basic)

867  return make_error_code(error::requires_64bit);
868  }

905  lib::error_code prepare_control(frame::opcode::value op,
906  std::string const & payload, message_ptr out) const
907  {
908  if (!out) {
909  return make_error_code(error::invalid_arguments);
910  }
911 
912  if (!frame::opcode::is_control(op)) {
913  return make_error_code(error::invalid_opcode);
914  }
915 
916  if (payload.size() > frame::limits::payload_size_basic) {
917  return make_error_code(error::control_too_big);
918  }
919 
920  frame::masking_key_type key;
921  bool masked = !base::m_server;
922 
923  frame::basic_header h(op,payload.size(),true,masked);
924 
925  std::string & o = out->get_raw_payload();
926  o.resize(payload.size());
927 
928  if (masked) {
929  // Generate masking key.
930  key.i = m_rng();
931 
932  frame::extended_header e(payload.size(),key.i);
933  out->set_header(frame::prepare_header(h,e));
934  this->masked_copy(payload,o,key);
935  } else {
936  frame::extended_header e(payload.size());
937  out->set_header(frame::prepare_header(h,e));
938  std::copy(payload.begin(),payload.end(),o.begin());
939  }
940 
941  out->set_prepared(true);
942 
943  return lib::error_code();
944  }
