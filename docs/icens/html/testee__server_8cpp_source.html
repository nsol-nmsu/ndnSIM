88 int main(int argc, char * argv[]) {
89  // Create a server endpoint
90  server testee_server;
91 
92  short port = 9002;
93  size_t num_threads = 1;
94 
95  if (argc == 3) {
96  port = atoi(argv[1]);
97  num_threads = atoi(argv[2]);
98  }
99 
100  try {
101  // Total silence
102  testee_server.clear_access_channels(websocketpp::log::alevel::all);
103  testee_server.clear_error_channels(websocketpp::log::alevel::all);
104 
105  // Initialize ASIO
106  testee_server.init_asio();
107  testee_server.set_reuse_addr(true);
108 
109  // Register our message handler
110  testee_server.set_message_handler(bind(&on_message,&testee_server,::_1,::_2));
111  testee_server.set_socket_init_handler(bind(&on_socket_init,::_1,::_2));
112 
113  // Listen on specified port with extended listen backlog
114  testee_server.set_listen_backlog(8192);
115  testee_server.listen(port);
116 
117  // Start the server accept loop
118  testee_server.start_accept();
119 
120  // Start the ASIO io_service run loop
121  if (num_threads == 1) {
122  testee_server.run();
123  } else {
124  typedef websocketpp::lib::shared_ptr<websocketpp::lib::thread> thread_ptr;
125  std::vector<thread_ptr> ts;
126  for (size_t i = 0; i < num_threads; i++) {
127  ts.push_back(websocketpp::lib::make_shared<websocketpp::lib::thread>(&server::run, &testee_server));
128  }
129 
130  for (size_t i = 0; i < num_threads; i++) {
131  ts[i]->join();
132  }
133  }
134 
135  } catch (websocketpp::exception const & e) {
136  std::cout << "exception: " << e.what() << std::endl;
137  }
138 }
