94  // Basic getting and setting

96  impl.setTpmLocator("tpmLocator");
97  BOOST_CHECK_EQUAL(impl.getTpmLocator(), "tpmLocator");
98 
99  // Add cert, and do not change tpmLocator

102  BOOST_CHECK(impl.hasKey(identity, keyId));
103  BOOST_CHECK(impl.hasCertificate(certName));
104 
105  // Set tpmLocator with the existing value, nothing should change.
106  impl.setTpmLocator("tpmLocator");

108  BOOST_CHECK(impl.hasKey(identity, keyId));
109  BOOST_CHECK(impl.hasCertificate(certName));
